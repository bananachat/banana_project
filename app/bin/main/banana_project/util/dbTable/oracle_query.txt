CREATE TABLE TB_USER
(
	id	VARCHAR2(32)	NOT NULL	PRIMARY KEY,
	password	VARCHAR2(256)	NOT NULL,
	name	VARCHAR2(32)	NOT NULL,
	hp	VARCHAR2(32)	UNIQUE,
	mail	VARCHAR2(32),
	nickname		VARCHAR2(32),
	stat	NUMBER(32)	NOT NULL,
	ins_date	DATE	DEFAULT	SYSDATE,
	upd_date	DATE	DEFAULT	SYSDATE,
	login_date	DATE	DEFAULT	SYSDATE
);


CREATE TABLE TB_CHAT_LIST
(
	no	NUMBER(32)	NOT NULL	PRIMARY KEY,
	chat_title	VARCHAR2(32)	DEFAULT	새 채팅',
	chat_date	DATE	DEFAULT SYSDATE,
	user_cnt	NUMBER(32),
	user_list	VARCHAR2(256),
	user_id	VARCHAR2(32)	UNIQUE
);


CREATE TABLE TB_CHAT_CONTANTS
(
	no	NUMBER(32)	NOT NULL	PRIMARY KEY,
	chat_title	VARCHAR2(32)	DEFAULT	새 채팅',
	user_list	VARCHAR2(32)	NOT NULL,
	contants	VARCHAR2(512),
	chat_date	DATE	DEFAULT SYSDATE,
	user_cnt	NUMBER(32)
);


CREATE TABLE TB_FRIENDS_LIST
(
	user_id	NUMBER(32)	NOT NULL	PRIMARY KEY,
	f_id	VARCHAR2(32)	NOT NULL,
	f_name	VARCHAR2(32)	NOT NULL,
	f_hp	VARCHAR2(32),
	f_mail	VARCHAR2(32)
);


CREATE TABLE TB_LOGIN_LOG
(
	login_date	DATE	DEFAULT SYSDATE	PRIMARY KEY	NOT NULL,
	protocol	NUMBER(32)	NOT NULL,
	fail_cnt	NUMBER(32)	DEFAULT	0,
	comments	VARCHAR2(256)
);


CREATE TABLE TB_CHAT_LOG
(
	chat_date	DATE	DEFAULT SYSDATE	PRIMARY KEY	NOT NULL,
	protocol	NUMBER(32)	NOT NULL,
	user_id	VARCHAR2(32)	NOT NULL,
	comments	VARCHAR2(256)
);
